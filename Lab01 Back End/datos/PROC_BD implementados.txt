Metodos de la BD:

PROFESOR:

Proc:
	INSERTAR_PROFESOR(CEDULA, NOMBRE, TELEFONO, EMAIL);
	MODIFICAR_PROFESOR(CEDULA, NOMBRE, TELEFONO, EMAIL);
	ELIMINAR_PROFESOR(CEDULA);

Func:
	BUSCAR_PROFESOR(CEDULA); 
	LISTAR_PROFESOR;

CURSO:

Proc:
	INSERTAR_CURSO(CODIGO, NOMBRE, CREDITO, HORAS);
	MODIFICAR_CURSO(CODIGO, NOMBRE, CREDITO, HORAS);
	ELIMINAR_CURSO(CODIGO);

Func:
	BUSCAR_CURSO(CODIGO);
	LISTAR_CURSO;

GRUPO: 

Proc:
	CREAR_GRUPO(CURSO, PROFE); -> Asigna el numero de grupo automaticamente
	ELIMINAR_GRUPO(CURSO, PROFE);
